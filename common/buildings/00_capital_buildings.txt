		script = buildings/on_all_capital_buildings

	# Undercoat inline script
	inline_script = {
		script = stu/jobs_add
		job_type = enforcers
		AMOUNT = 1
		trigger = "
			owner = { has_dystopian_society = yes }
		"
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_dystopian_society = yes } # undercoat: moved to scripted trigger
		}
		job_colonist_add = -1
	}

	triggered_planet_modifier = {
		potential = {

	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_individual_machine = yes }
		}
		job_colonist_add = -1
	}

	# Undercoat inline script
	inline_script = {
		script = stu/jobs_add
		job_type = roboticists
		AMOUNT = 1
		trigger = "
			owner = { has_active_tradition = tr_synthetics_prefabricated_components }
		"
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_active_tradition = tr_synthetics_prefabricated_components }
		}
		job_colonist_add = -1
	}

		category = planet_buildings

			energy = 1
