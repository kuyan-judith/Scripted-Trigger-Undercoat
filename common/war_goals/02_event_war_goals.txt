					if = {
						limit = {
							any_war = {

									prev = {
										using_war_goal = {
											type = wg_war_in_heaven
											owner = prev
										}
									}
								}
							}

						}
					}
				}
			}
			add_opinion_modifier = {
				who = from
				modifier = opinion_tried_to_secede
			}
			from = {
				add_opinion_modifier = {
					who = root
