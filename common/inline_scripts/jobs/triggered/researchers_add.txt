triggered_planet_modifier = {
	$mult_line$
	potential = {
		OR = {
			is_scope_valid = no
			AND = {
				exists = owner
				$trigger$
				owner = { is_regular_empire = yes }
			}
		}
	}
	modifier = {
		job_researcher_add = $AMOUNT$
	}
}
